security_check() {
    clear
    
    # Цвета
    RED='\033[0;31m'
    GREEN='\033[0;32m'
    YELLOW='\033[1;33m'
    BLUE='\033[0;34m'
    PURPLE='\033[0;35m'
    CYAN='\033[0;36m'
    WHITE='\033[1;37m'
    NC='\033[0m'
    BOLD='\033[1m'

    echo -e "${CYAN}${BOLD}"
    echo "╔══════════════════════════════════════════════════════════════════════════════════╗"
    echo "║                      🔒 SECURITY CHECK — ORACLE BOT v16.0                        ║"
    echo "╚══════════════════════════════════════════════════════════════════════════════════╝"
    echo -e "${NC}"

    # Функция для отрисовки строки
    draw_line() {
        echo -e "${CYAN}║${NC} $1"
    }

    # Функция для статуса
    status_icon() {
        if [ "$1" == "ok" ]; then
            echo -e "${GREEN}✅${NC}"
        elif [ "$1" == "warn" ]; then
            echo -e "${YELLOW}⚠️${NC}"
        elif [ "$1" == "error" ]; then
            echo -e "${RED}❌${NC}"
        elif [ "$1" == "info" ]; then
            echo -e "${BLUE}ℹ️${NC}"
        fi
    }

    # ========== 1. БАЗЫ ДАННЫХ ==========
    echo -e "${CYAN}╔══════════════════════════════════════════════════════════════════════════════╗${NC}"
    echo -e "${CYAN}║${WHITE}${BOLD}                        📁 БАЗЫ ДАННЫХ                                    ${CYAN}║${NC}"
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"

    if [ -f ~/oracle-bot/oracle_bot.db ]; then
        PERMS=$(stat -c "%a" ~/oracle-bot/oracle_bot.db 2>/dev/null)
        if [ "$PERMS" = "600" ]; then
            echo -e "${CYAN}║${NC}   $(status_icon ok) oracle_bot.db  ${GREEN}✓ 600 (безопасно)${NC} $(printf '%*s' $((40-${#PERMS})) '')${CYAN}║${NC}"
        else
            echo -e "${CYAN}║${NC}   $(status_icon error) oracle_bot.db  ${RED}✗ $PERMS (должно быть 600)${NC}${CYAN}║${NC}"
        fi
    else
        echo -e "${CYAN}║${NC}   $(status_icon warn) oracle_bot.db  ${YELLOW}⚠ не найден${NC}${CYAN}║${NC}"
    fi

    if [ -f ~/oracle-bot/web_panel.db ]; then
        PERMS=$(stat -c "%a" ~/oracle-bot/web_panel.db 2>/dev/null)
        if [ "$PERMS" = "600" ]; then
            echo -e "${CYAN}║${NC}   $(status_icon ok) web_panel.db   ${GREEN}✓ 600 (безопасно)${NC}${CYAN}║${NC}"
        else
            echo -e "${CYAN}║${NC}   $(status_icon error) web_panel.db   ${RED}✗ $PERMS (должно быть 600)${NC}${CYAN}║${NC}"
        fi
    else
        echo -e "${CYAN}║${NC}   $(status_icon warn) web_panel.db   ${YELLOW}⚠ не найден${NC}${CYAN}║${NC}"
    fi

    # ========== 2. ЛОГИ ==========
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"
    echo -e "${CYAN}║${WHITE}${BOLD}                        📊 ЛОГИ СИСТЕМЫ                                     ${CYAN}║${NC}"
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"

    if [ -d ~/oracle-bot/logs ]; then
        LOGS_PERMS=$(stat -c "%a" ~/oracle-bot/logs 2>/dev/null)
        if [ "$LOGS_PERMS" = "700" ]; then
            echo -e "${CYAN}║${NC}   $(status_icon ok) папка logs     ${GREEN}✓ 700 (безопасно)${NC}${CYAN}║${NC}"
        else
            echo -e "${CYAN}║${NC}   $(status_icon error) папка logs     ${RED}✗ $LOGS_PERMS (должно быть 700)${NC}${CYAN}║${NC}"
        fi

        # Проверка файлов логов
        BAD_LOGS=0
        for log in ~/oracle-bot/logs/*.log; do
            if [ -f "$log" ]; then
                PERMS=$(stat -c "%a" "$log" 2>/dev/null)
                if [ "$PERMS" != "600" ] && [ "$PERMS" != "640" ]; then
                    echo -e "${CYAN}║${NC}   $(status_icon error) $(basename $log) ${RED}✗ $PERMS${NC}${CYAN}║${NC}"
                    BAD_LOGS=1
                fi
            fi
        done
        if [ $BAD_LOGS -eq 0 ]; then
            echo -e "${CYAN}║${NC}   $(status_icon ok) все логи       ${GREEN}✓ права в порядке${NC}${CYAN}║${NC}"
        fi
    else
        echo -e "${CYAN}║${NC}   $(status_icon warn) папка logs     ${YELLOW}⚠ не найдена${NC}${CYAN}║${NC}"
    fi

    # ========== 3. ТОКЕНЫ ==========
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"
    echo -e "${CYAN}║${WHITE}${BOLD}                        🔑 ФАЙЛЫ С ТОКЕНАМИ                                 ${CYAN}║${NC}"
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"

    if [ -f ~/oracle-bot/.env.secure ]; then
        PERMS=$(stat -c "%a" ~/oracle-bot/.env.secure 2>/dev/null)
        if [ "$PERMS" = "600" ]; then
            echo -e "${CYAN}║${NC}   $(status_icon ok) .env.secure    ${GREEN}✓ 600 (безопасно)${NC}${CYAN}║${NC}"
        else
            echo -e "${CYAN}║${NC}   $(status_icon error) .env.secure    ${RED}✗ $PERMS (должно быть 600)${NC}${CYAN}║${NC}"
        fi
    else
        echo -e "${CYAN}║${NC}   $(status_icon info) .env.secure    ${BLUE}ℹ не используется (или другое имя)${NC}${CYAN}║${NC}"
    fi

    # ========== 4. СТАТУС ПРОЦЕССОВ ==========
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"
    echo -e "${CYAN}║${WHITE}${BOLD}                        ⚙️ СТАТУС ПРОЦЕССОВ                                   ${CYAN}║${NC}"
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"

    if pgrep -f "web_panel.py" > /dev/null; then
        WEB_PID=$(pgrep -f "web_panel.py" | head -1)
        echo -e "${CYAN}║${NC}   $(status_icon ok) веб-панель     ${GREEN}✓ запущена (PID: $WEB_PID)${NC}${CYAN}║${NC}"
    else
        echo -e "${CYAN}║${NC}   $(status_icon warn) веб-панель     ${YELLOW}⚠ не запущена${NC}${CYAN}║${NC}"
    fi

    if pgrep -f "oracle_bot_final_complete.py" > /dev/null; then
        BOT_PID=$(pgrep -f "oracle_bot_final_complete.py" | head -1)
        echo -e "${CYAN}║${NC}   $(status_icon ok) бот            ${GREEN}✓ запущен (PID: $BOT_PID)${NC}${CYAN}║${NC}"
    else
        echo -e "${CYAN}║${NC}   $(status_icon warn) бот            ${YELLOW}⚠ не запущен${NC}${CYAN}║${NC}"
    fi

    # ========== 5. ПОРТЫ ==========
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"
    echo -e "${CYAN}║${WHITE}${BOLD}                        🌐 ОТКРЫТЫЕ ПОРТЫ                                   ${CYAN}║${NC}"
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"

    if command -v netstat &> /dev/null; then
        PORTS=$(netstat -tuln 2>/dev/null | grep -E ":(5000|22|80|443)" | awk '{print $4}' | cut -d: -f2 | sort -u | tr '\n' ' ')
        if [ -n "$PORTS" ]; then
            echo -e "${CYAN}║${NC}   $(status_icon info) открытые порты: ${BLUE}$PORTS${NC}${CYAN}║${NC}"
            if echo "$PORTS" | grep -q "5000"; then
                echo -e "${CYAN}║${NC}   $(status_icon warn) порт 5000 открыт ${YELLOW}⚠ только для локального доступа${NC}${CYAN}║${NC}"
            fi
        else
            echo -e "${CYAN}║${NC}   $(status_icon ok) нет открытых портов${GREEN} ✓${NC}${CYAN}║${NC}"
        fi
    else
        echo -e "${CYAN}║${NC}   $(status_icon info) netstat не найден${NC}${CYAN}║${NC}"
    fi

    # ========== 6. ИТОГ ==========
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"
    echo -e "${CYAN}║${WHITE}${BOLD}                        📋 ИТОГ ПРОВЕРКИ                                      ${CYAN}║${NC}"
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════════════════╣${NC}"

    echo -e "${CYAN}║${NC}   $(status_icon ok) ${GREEN}Базы данных защищены${NC}                                  ${CYAN}║${NC}"
    echo -e "${CYAN}║${NC}   $(status_icon ok) ${GREEN}Логи защищены${NC}                                          ${CYAN}║${NC}"
    echo -e "${CYAN}║${NC}   $(status_icon info) ${BLUE}Рекомендации:${NC}                                         ${CYAN}║${NC}"
    echo -e "${CYAN}║${NC}      • Не открывай порт 5000 наружу (используй localhost)                ${CYAN}║${NC}"
    echo -e "${CYAN}║${NC}      • Регулярно чисти старые лозы                                       ${CYAN}║${NC}"
    echo -e "${CYAN}║${NC}      • Следи за правами доступа к файлам                                 ${CYAN}║${NC}"
    echo -e "${CYAN}║${NC}      • Используй HTTPS для веб-панели (самоподписанный сертификат)       ${CYAN}║${NC}"

    echo -e "${CYAN}╚══════════════════════════════════════════════════════════════════════════════╝${NC}"
    echo ""
}

# Умный пуш
gpush() {
    echo -e "\n\033[0;35m🔮 ORACLE GIT PUSHER\033[0m"
    echo -e "\033[0;33m📁 $(pwd | sed "s|$HOME|~|")\033[0m\n"

    if [[ -n $(git status -s) ]]; then
        echo -e "\033[0;33m📝 Изменения:\033[0m"
        git status -s
        echo ""
        read -p "Сделать коммит? (y/n): " commit_ans
        if [[ "$commit_ans" == "y" || "$commit_ans" == "Y" ]]; then
            echo -n "💬 Сообщение: "
            read msg
            git add .
            git commit -m "$msg"
            echo -e "\033[0;35m✅ Коммит сделан\033[0m\n"
        fi
    fi

    echo -e "\033[0;35m📤 Пуш...\033[0m"
    if git push origin main 2>&1 | grep -q "rejected"; then
        echo -e "\033[0;31m⚠️ Конфликт\033[0m"
        read -p "Сделать pull --rebase? (y/n): " pull_ans
        if [[ "$pull_ans" == "y" || "$pull_ans" == "Y" ]]; then
            git pull --rebase origin main && git push origin main && echo -e "\033[0;35m🎉 Успешно!\033[0m" || echo -e "\033[0;31m❌ Ошибка\033[0m"
        fi
    else
        echo -e "\033[0;35m🎉 Успешно!\033[0m"
    fi
}

# Статус
gss() {
    echo -e "\n\033[0;35m🔮 ORACLE STATUS\033[0m"
    echo -e "\033[0;33m📁 $(pwd | sed "s|$HOME|~|")\033[0m"
    echo -e "\033[0;35m$(git status -s || echo '✨ Чисто')\033[0m"
    echo -e "\n\033[0;33m📜 Последние коммиты:\033[0m"
    git log --oneline --decorate -5
}
